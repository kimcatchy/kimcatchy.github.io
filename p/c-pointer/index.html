#include <stdio.h>

void swap(int* a, int* b) {
    int tmp;
    tmp = *a;
    *a = *b;
    *b = tmp;
    return;
}

int main(void) {
    int i, j, a[10];
    int min, idx;

    for (i = 0; i < 10; i++) {
        scanf_s("%d", &a[i]);
    }

    for (i = 0; i < 9; i++) {
        min = a[i];
        idx = i;
        for (j = i; j < 10; j++) {
            if (a[j] < min) {
                min = a[j];
                idx = j;
            }
        }
        swap(&a[i], &a[idx]);
    }

    for (i = 0; i < 10; i++) {
        printf("%d\\n", a[i]);
    }
}
