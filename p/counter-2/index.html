/**
 * @param {integer} init
 * @return { increment: Function, decrement: Function, reset: Function }
 */
var createCounter = function(init) {
    let i = init
    return {
        increment: function() {
            return ++i
        },
        decrement: function() {
            return --i
        },
        reset: function() {
            return i = init
        }
    }
};
